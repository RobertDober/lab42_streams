A sketch for the merge_by method

  def merge_by *streams, &beh
     heads = streams
       .reject( :empty? )
       .map( :head )
       .sort_by( beh )

     return empty_stream if heads.empty?

     cons_stream_n_heads heads do
        merge_by( streams.map( :tail ), &beh )
     end
  end
